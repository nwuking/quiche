73// 4.3.1.1.  Startup Dynamics
74fn bbr2_check_startup_done(r: &mut Congestion) {
75    bbr2_check_startup_full_bandwidth(r);
76    bbr2_check_startup_high_loss(r);
77
78    if r.bbr2_state.state == BBR2StateMachine::Startup && r.bbr2_state.filled_pipe
79    {
80        bbr2_enter_drain(r);
81    }
82}
83
84// 4.3.1.2.  Exiting Startup Based on Bandwidth Plateau
85fn bbr2_check_startup_full_bandwidth(r: &mut Congestion) {
86    if r.bbr2_state.filled_pipe ||
87        !r.bbr2_state.round_start ||
88        r.delivery_rate.sample_is_app_limited()
89    {
90        // No need to check for a full pipe now.
91        return;
92    }
93
94    // Still growing?
95    if r.bbr2_state.max_bw >=
96        (r.bbr2_state.full_bw as f64 * MAX_BW_GROWTH_THRESHOLD) as u64
97    {
98        // Record new baseline level
99        r.bbr2_state.full_bw = r.bbr2_state.max_bw;
100        r.bbr2_state.full_bw_count = 0;
101        return;
102    }
103
104    // Another round w/o much growth
105    r.bbr2_state.full_bw_count += 1;
106
107    if r.bbr2_state.full_bw_count >= MAX_BW_COUNT {
108        r.bbr2_state.filled_pipe = true;
109    }
110}

112// 4.3.1.3.  Exiting Startup Based on Packet Loss
113fn bbr2_check_startup_high_loss(r: &mut Congestion) {
114    // TODO: this is not implemented (not in the draft)
115    if r.bbr2_state.loss_round_start &&
116        r.bbr2_state.in_recovery &&
117        r.bbr2_state.loss_events_in_round >= FULL_LOSS_COUNT as usize &&
118        per_loss::bbr2_is_inflight_too_high(r)
119    {
120        bbr2_handle_queue_too_high_in_startup(r);
121    }
122    if r.bbr2_state.loss_round_start {
123        r.bbr2_state.loss_events_in_round = 0
124    }

127fn bbr2_handle_queue_too_high_in_startup(r: &mut Congestion) {
128    r.bbr2_state.filled_pipe = true;
129    r.bbr2_state.inflight_hi = bbr2_inflight(r, r.bbr2_state.max_bw, 1.0);

132// 4.3.2.  Drain
133fn bbr2_enter_drain(r: &mut Congestion) {
134    let bbr = &mut r.bbr2_state;
135
136    bbr.state = BBR2StateMachine::Drain;
137
138    // pace slowly
139    bbr.pacing_gain = PACING_GAIN / STARTUP_CWND_GAIN;
140
141    // maintain cwnd
142    bbr.cwnd_gain = STARTUP_CWND_GAIN;
143}

145fn bbr2_check_drain(r: &mut Congestion, in_flight: usize, now: Instant) {
146    if r.bbr2_state.state == BBR2StateMachine::Drain &&
147        in_flight <= bbr2_inflight(r, r.bbr2_state.max_bw, 1.0)
148    {
149        // BBR estimates the queue was drained
150        bbr2_enter_probe_bw(r, now);
151    }
152}

154// 4.3.3.  ProbeBW
155// 4.3.3.5.3.  Design Considerations for Choosing Constant Parameters
156fn bbr2_check_time_to_probe_bw(r: &mut Congestion, now: Instant) -> bool {
157    // Is it time to transition from DOWN or CRUISE to REFILL?
158    if bbr2_has_elapsed_in_phase(r, r.bbr2_state.bw_probe_wait, now) ||
159        bbr2_is_reno_coexistence_probe_time(r)
160    {
161        bbr2_start_probe_bw_refill(r);
162
163        return true;
164    }
165
166    false
167}

169// Randomized decision about how long to wait until
170// probing for bandwidth, using round count and wall clock.
171fn bbr2_pick_probe_wait(r: &mut Congestion) {
172    let bbr = &mut r.bbr2_state;
173
174    // Decide random round-trip bound for wait
175    bbr.rounds_since_probe = rand::rand_u8() as usize % 2;
176
177    // Decide the random wall clock bound for wait
178    bbr.bw_probe_wait = Duration::from_secs_f64(
179        2.0 + rand::rand_u64_uniform(1000000) as f64 / 1000000.0,
180    );
181}
182
183fn bbr2_is_reno_coexistence_probe_time(r: &mut Congestion) -> bool {
184    let reno_rounds = bbr2_target_inflight(r);
185    let rounds = reno_rounds.min(63);
186
187    r.bbr2_state.rounds_since_probe >= rounds
188}

192pub fn bbr2_target_inflight(r: &mut Congestion) -> usize {
193    r.bbr2_state.bdp.min(r.congestion_window)

201pub fn bbr2_start_probe_bw_down(r: &mut Congestion, now: Instant) {
202    per_loss::bbr2_reset_congestion_signals(r);
203
204    // not growing inflight_hi
205    r.bbr2_state.probe_up_cnt = usize::MAX;
206
207    bbr2_pick_probe_wait(r);
208
209    // start wall clock
210    r.bbr2_state.cycle_stamp = now;
211    r.bbr2_state.ack_phase = BBR2AckPhase::ProbeStopping;
212
213    bbr2_start_round(r);
214
215    r.bbr2_state.state = BBR2StateMachine::ProbeBWDOWN;
216    r.bbr2_state.pacing_gain = PROBE_DOWN_PACING_GAIN;
217    r.bbr2_state.cwnd_gain = CWND_GAIN
218}

220fn bbr2_start_probe_bw_cruise(r: &mut Congestion) {
221    let bbr = &mut r.bbr2_state;
222
223    bbr.state = BBR2StateMachine::ProbeBWCRUISE;
224    bbr.pacing_gain = PACING_GAIN;
225    bbr.cwnd_gain = CWND_GAIN;
226}
227
228fn bbr2_start_probe_bw_refill(r: &mut Congestion) {
229    per_loss::bbr2_reset_lower_bounds(r);
230
231    r.bbr2_state.bw_probe_up_rounds = 0;
232    r.bbr2_state.bw_probe_up_acks = 0;
233    r.bbr2_state.ack_phase = BBR2AckPhase::Refilling;
234
235    bbr2_start_round(r);
236
237    r.bbr2_state.state = BBR2StateMachine::ProbeBWREFILL;
238    r.bbr2_state.pacing_gain = PACING_GAIN;
239    r.bbr2_state.cwnd_gain = CWND_GAIN;
240}

242fn bbr2_start_probe_bw_up(r: &mut Congestion, now: Instant) {
243    r.bbr2_state.ack_phase = BBR2AckPhase::ProbeStarting;
244
245    bbr2_start_round(r);
246
247    // Start wall clock.
248    r.bbr2_state.cycle_stamp = now;
249    r.bbr2_state.state = BBR2StateMachine::ProbeBWUP;
250    r.bbr2_state.pacing_gain = PROBE_UP_PACING_GAIN;
251    r.bbr2_state.cwnd_gain = CWND_GAIN;
252
253    bbr2_raise_inflight_hi_slope(r);
254}

256// The core state machine logic for ProbeBW
257fn bbr2_update_probe_bw_cycle_phase(
258    r: &mut Congestion, in_flight: usize, now: Instant,
259) {
260    if !r.bbr2_state.filled_pipe {
261        // only handling steady-state behavior here
262        return;
263    }
264
265    bbr2_adapt_upper_bounds(r, now);
266
267    if !bbr2_is_in_a_probe_bw_state(r) {
268        // only handling ProbeBW states here
269        return;
270    }
271
272    match r.bbr2_state.state {
273        BBR2StateMachine::ProbeBWDOWN => {
274            if bbr2_check_time_to_probe_bw(r, now) {
275                // Already decided state transition.
276                return;
277            }
278
279            if bbr2_check_time_to_cruise(r, in_flight) {
280                bbr2_start_probe_bw_cruise(r);
281            }
282        },
283
284        BBR2StateMachine::ProbeBWCRUISE => {
285            bbr2_check_time_to_probe_bw(r, now);
286        },
287
288        BBR2StateMachine::ProbeBWREFILL => {
289            // After one round of REFILL, start UP.
290            if r.bbr2_state.round_start {
291                r.bbr2_state.bw_probe_samples = true;
292
293                bbr2_start_probe_bw_up(r, now);
294            }
295        },
296
297        BBR2StateMachine::ProbeBWUP => {
298            if bbr2_has_elapsed_in_phase(r, r.bbr2_state.min_rtt, now) &&
299                in_flight > bbr2_inflight(r, r.bbr2_state.max_bw, 1.25)
300            {
301                bbr2_start_probe_bw_down(r, now);
302            }
303        },
304
305        _ => (),
306    }
307}

309pub fn bbr2_is_in_a_probe_bw_state(r: &mut Congestion) -> bool {
310    let state = r.bbr2_state.state;
311
312    state == BBR2StateMachine::ProbeBWDOWN ||
313        state == BBR2StateMachine::ProbeBWCRUISE ||
314        state == BBR2StateMachine::ProbeBWREFILL ||
315        state == BBR2StateMachine::ProbeBWUP
316}

318fn bbr2_check_time_to_cruise(r: &mut Congestion, in_flight: usize) -> bool {
319    if in_flight > bbr2_inflight_with_headroom(r) {
320        // Not enough headroom.
321        return false;
322    }
323
324    if in_flight <= bbr2_inflight(r, r.bbr2_state.max_bw, 1.0) {
325        // inflight <= estimated BDP
326        return true;
327    }
328
329    false
330}

332fn bbr2_has_elapsed_in_phase(
333    r: &mut Congestion, interval: Duration, now: Instant,
334) -> bool {
335    now > r.bbr2_state.cycle_stamp + interval
336}

338// Return a volume of data that tries to leave free
339// headroom in the bottleneck buffer or link for
340// other flows, for fairness convergence and lower
341// RTTs and loss
342fn bbr2_inflight_with_headroom(r: &mut Congestion) -> usize {
343    let bbr = &mut r.bbr2_state;
344
345    if bbr.inflight_hi == usize::MAX {
346        return usize::MAX;
347    }
348
349    let headroom = ((HEADROOM * bbr.inflight_hi as f64) as usize).max(1);
350
351    bbr.inflight_hi
352        .saturating_sub(headroom)
353        .max(bbr2_min_pipe_cwnd(r))
354}

356// Raise inflight_hi slope if appropriate.
357fn bbr2_raise_inflight_hi_slope(r: &mut Congestion) {
358    let bbr = &mut r.bbr2_state;
359
360    let growth_this_round = (1 << bbr.bw_probe_up_rounds) * r.max_datagram_size;
361
362    bbr.bw_probe_up_rounds = (bbr.bw_probe_up_rounds + 1).min(30);
363    bbr.probe_up_cnt = (r.congestion_window / growth_this_round).max(1);
364}
365
366// Increase inflight_hi if appropriate.
367fn bbr2_probe_inflight_hi_upward(r: &mut Congestion) {
368    if r.app_limited || r.congestion_window < r.bbr2_state.inflight_hi {
369        // Not fully using inflight_hi, so don't grow it.
370        return;
371    }
372
373    let bbr = &mut r.bbr2_state;
374
375    // bw_probe_up_acks is a packet count.
376    bbr.bw_probe_up_acks += 1;
377
378    if bbr.bw_probe_up_acks >= bbr.probe_up_cnt {
379        let delta = bbr.bw_probe_up_acks / bbr.probe_up_cnt;
380
381        bbr.bw_probe_up_acks -= delta * bbr.probe_up_cnt;
382
383        bbr.inflight_hi += delta * r.max_datagram_size;
384    }
385
386    if bbr.round_start {
387        bbr2_raise_inflight_hi_slope(r);
388    }
389}

391// Track ACK state and update bbr.max_bw window and
392// bbr.inflight_hi and bbr.bw_hi.
393fn bbr2_adapt_upper_bounds(r: &mut Congestion, now: Instant) {
394    if r.bbr2_state.ack_phase == BBR2AckPhase::ProbeStarting &&
395        r.bbr2_state.round_start
396    {
397        // Starting to get bw probing samples.
398        r.bbr2_state.ack_phase = BBR2AckPhase::ProbeFeedback;
399    }
400
401    if r.bbr2_state.ack_phase == BBR2AckPhase::ProbeStopping &&
402        r.bbr2_state.round_start
403    {
404        r.bbr2_state.bw_probe_samples = false;
405        r.bbr2_state.ack_phase = BBR2AckPhase::Init;
406
407        // End of samples from bw probing phase.
408        if bbr2_is_in_a_probe_bw_state(r) &&
409            !r.delivery_rate.sample_is_app_limited()
410        {
411            bbr2_advance_max_bw_filter(r);
412        }
413    }
414
415    if !per_loss::bbr2_check_inflight_too_high(r, now) {
416        // Loss rate is safe. Adjust upper bounds upward.
417        if r.bbr2_state.inflight_hi == usize::MAX ||
418            r.bbr2_state.bw_hi == u64::MAX
419        {
420            // No upper bounds to raise.
421            return;
422        }
423
424        if r.bbr2_state.tx_in_flight > r.bbr2_state.inflight_hi {
425            r.bbr2_state.inflight_hi = r.bbr2_state.tx_in_flight;
426        }
427
428        // TODO: what's rs.bw???
429        if r.delivery_rate() > r.bbr2_state.bw_hi {
430            r.bbr2_state.bw_hi = r.delivery_rate();
431        }
432
433        if r.bbr2_state.state == BBR2StateMachine::ProbeBWUP {
434            bbr2_probe_inflight_hi_upward(r);
435        }
436    }
437}

440// 4.3.4.4.  ProbeRTT Logic
441fn bbr2_update_min_rtt(r: &mut Congestion, now: Instant) {
442    let bbr = &mut r.bbr2_state;
443
444    bbr.probe_rtt_expired = now > bbr.probe_rtt_min_stamp + PROBE_RTT_INTERVAL;
445
446    let rs_rtt = r.delivery_rate.sample_rtt();
447
448    if !rs_rtt.is_zero() &&
449        (rs_rtt < bbr.probe_rtt_min_delay || bbr.probe_rtt_expired)
450    {
451        bbr.probe_rtt_min_delay = rs_rtt;
452        bbr.probe_rtt_min_stamp = now;
453    }
454
455    let min_rtt_expired =
456        now > bbr.min_rtt_stamp + rs_rtt.saturating_mul(MIN_RTT_FILTER_LEN);
457
458    // To do: Figure out Probe RTT logic
459    // if bbr.probe_rtt_min_delay < bbr.min_rtt ||  bbr.min_rtt == INITIAL_RTT ||
460    // min_rtt_expired {
461    if bbr.min_rtt == rtt::INITIAL_RTT || min_rtt_expired {
462        // bbr.min_rtt = bbr.probe_rtt_min_delay;
463        // bbr.min_rtt_stamp = bbr.probe_rtt_min_stamp;
464        bbr.min_rtt = rs_rtt;
465        bbr.min_rtt_stamp = now;
466    }
467}

469fn bbr2_check_probe_rtt(r: &mut Congestion, in_flight: usize, now: Instant) {
470    if r.bbr2_state.state != BBR2StateMachine::ProbeRTT &&
471        r.bbr2_state.probe_rtt_expired &&
472        !r.bbr2_state.idle_restart
473    {
474        bbr2_enter_probe_rtt(r);
475
476        r.bbr2_state.prior_cwnd = per_ack::bbr2_save_cwnd(r);
477        r.bbr2_state.probe_rtt_done_stamp = None;
478        r.bbr2_state.ack_phase = BBR2AckPhase::ProbeStopping;
479
480        bbr2_start_round(r);
481    }
482
483    if r.bbr2_state.state == BBR2StateMachine::ProbeRTT {
484        bbr2_handle_probe_rtt(r, in_flight, now);
485    }
486
487    if r.delivery_rate.sample_delivered() > 0 {
488        r.bbr2_state.idle_restart = false;
489    }
490}

492fn bbr2_enter_probe_rtt(r: &mut Congestion) {
493    let bbr = &mut r.bbr2_state;
494
495    bbr.state = BBR2StateMachine::ProbeRTT;
496    bbr.pacing_gain = PACING_GAIN;
497    bbr.cwnd_gain = PROBE_RTT_CWND_GAIN;
498}
499
500fn bbr2_handle_probe_rtt(r: &mut Congestion, in_flight: usize, now: Instant) {
501    // Ignore low rate samples during ProbeRTT.
502    r.delivery_rate.update_app_limited(true);
503
504    if r.bbr2_state.probe_rtt_done_stamp.is_some() {
505        if r.bbr2_state.round_start {
506            r.bbr2_state.probe_rtt_round_done = true;
507        }
508
509        if r.bbr2_state.probe_rtt_round_done {
510            bbr2_check_probe_rtt_done(r, now);
511        }
512    } else if in_flight <= bbr2_probe_rtt_cwnd(r) {
513        // Wait for at least ProbeRTTDuration to elapse.
514        r.bbr2_state.probe_rtt_done_stamp = Some(now + PROBE_RTT_DURATION);
515
516        // Wait for at lease one round to elapse.
517        r.bbr2_state.probe_rtt_round_done = false;
518
519        bbr2_start_round(r);
520    }
521}
522
523pub fn bbr2_check_probe_rtt_done(r: &mut Congestion, now: Instant) {
524    let bbr = &mut r.bbr2_state;
525
526    if let Some(probe_rtt_done_stamp) = bbr.probe_rtt_done_stamp {
527        if now > probe_rtt_done_stamp {
528            // Schedule next ProbeRTT.
529            bbr.probe_rtt_min_stamp = now;
530
531            bbr2_restore_cwnd(r);
532            bbr2_exit_probe_rtt(r, now);
533        }
534    }
535}

537// 4.3.4.5.  Exiting ProbeRTT
538fn bbr2_exit_probe_rtt(r: &mut Congestion, now: Instant) {
539    per_loss::bbr2_reset_lower_bounds(r);
540
541    if r.bbr2_state.filled_pipe {
542        bbr2_start_probe_bw_down(r, now);
543        bbr2_start_probe_bw_cruise(r);
544    } else {
545        init::bbr2_enter_startup(r);
546    }
547}

549// 4.5.1.  BBR.round_count: Tracking Packet-Timed Round Trips
550fn bbr2_update_round(r: &mut Congestion, packet: &Acked) {
551    if packet.delivered >= r.bbr2_state.next_round_delivered {
552        bbr2_start_round(r);
553
554        r.bbr2_state.round_count += 1;
555        r.bbr2_state.rounds_since_probe += 1;
556        r.bbr2_state.round_start = true;
557    } else {
558        r.bbr2_state.round_start = false;
559    }
560}
561
562fn bbr2_start_round(r: &mut Congestion) {
563    r.bbr2_state.next_round_delivered = r.delivery_rate.delivered();
564}

566// 4.5.2.4.  Updating the BBR.max_bw Max Filter
567pub fn bbr2_update_max_bw(r: &mut Congestion, packet: &Acked) {
568    bbr2_update_round(r, packet);
569
570    if r.delivery_rate() >= r.bbr2_state.max_bw ||
571        !r.delivery_rate.sample_is_app_limited()
572    {
573        let max_bw_filter_len = r
574            .delivery_rate
575            .sample_rtt()
576            .saturating_mul(MIN_RTT_FILTER_LEN);
577
578        r.bbr2_state.max_bw = r.bbr2_state.max_bw_filter.running_max(
579            max_bw_filter_len,
580            r.bbr2_state.start_time +
581                Duration::from_secs(r.bbr2_state.cycle_count),
582            r.delivery_rate(),
583        );
584    }
585}

597// 4.5.5.  BBR.extra_acked
598fn bbr2_update_ack_aggregation(r: &mut Congestion, packet: &Acked, now: Instant) {
599    let bbr = &mut r.bbr2_state;
600
601    // Find excess ACKed beyond expected amount over this interval.
602    let interval = now - bbr.extra_acked_interval_start;
603    let mut expected_delivered =
604        (bbr.bw as f64 * interval.as_secs_f64()) as usize;
605
606    // Reset interval if ACK rate is below expected rate.
607    if bbr.extra_acked_delivered <= expected_delivered {
608        bbr.extra_acked_delivered = 0;
609        bbr.extra_acked_interval_start = now;
610        expected_delivered = 0;
611    }
612
613    bbr.extra_acked_delivered += packet.size;
614
615    let extra = bbr.extra_acked_delivered.saturating_sub(expected_delivered);
616    let extra = extra.min(r.congestion_window);
617
618    let extra_acked_filter_len = r
619        .delivery_rate
620        .sample_rtt()
621        .saturating_mul(MIN_RTT_FILTER_LEN);
622
623    bbr.extra_acked = bbr.extra_acked_filter.running_max(
624        extra_acked_filter_len,
625        bbr.start_time + Duration::from_secs(bbr.round_count),
626        extra,
627    );
628}

630// 4.6.3.  Send Quantum: BBR.send_quantum
631fn bbr2_set_send_quantum(r: &mut Congestion) {
632    let bbr = &mut r.bbr2_state;
633
634    let rate = bbr.pacing_rate;
635    let floor = if rate < PACING_RATE_1_2MBPS {
636        r.max_datagram_size
637    } else {
638        2 * r.max_datagram_size
639    };
640
641    r.send_quantum = cmp::min((rate / 1000_u64) as usize, 64 * 1024); // Assumes send buffer is limited to 64KB
642    r.send_quantum = r.send_quantum.max(floor);
643}

645// 4.6.4.1.  Initial cwnd
646// 4.6.4.2.  Computing BBR.max_inflight
647fn bbr2_bdp_multiple(r: &mut Congestion, bw: u64, gain: f64) -> usize {
648    let bbr = &mut r.bbr2_state;
649
650    if bbr.min_rtt == Duration::MAX {
651        // No valid RTT samples yet.
652        return r.max_datagram_size * r.initial_congestion_window_packets;
653    }
654
655    bbr.bdp = (bw as f64 * bbr.min_rtt.as_secs_f64()) as usize;
656
657    (gain * bbr.bdp as f64) as usize
658}

660fn bbr2_quantization_budget(r: &mut Congestion, inflight: usize) -> usize {
661    bbr2_update_offload_budget(r);
662
663    let inflight = inflight.max(r.bbr2_state.offload_budget);
664    let inflight = inflight.max(bbr2_min_pipe_cwnd(r));
665
666    // TODO: cycle_idx is unused
667    if r.bbr2_state.state == BBR2StateMachine::ProbeBWUP {
668        return inflight + 2 * r.max_datagram_size;
669    }
670
671    inflight
672}

674fn bbr2_inflight(r: &mut Congestion, bw: u64, gain: f64) -> usize {
675    let inflight = bbr2_bdp_multiple(r, bw, gain);
676
677    bbr2_quantization_budget(r, inflight)
678}

688    r.bbr2_state.max_inflight = bbr2_quantization_budget(r, inflight);

691// 4.6.4.4.  Modulating cwnd in Loss Recovery
692pub fn bbr2_save_cwnd(r: &mut Congestion) -> usize {
693    if !r.bbr2_state.in_recovery &&
694        r.bbr2_state.state != BBR2StateMachine::ProbeRTT
695    {
696        r.congestion_window
697    } else {
698        r.congestion_window.max(r.bbr2_state.prior_cwnd)
699    }
700}
701
702pub fn bbr2_restore_cwnd(r: &mut Congestion) {
703    r.congestion_window = r.congestion_window.max(r.bbr2_state.prior_cwnd);
704}

706fn bbr2_modulate_cwnd_for_recovery(r: &mut Congestion, in_flight: usize) {
707    let acked_bytes = r.bbr2_state.newly_acked_bytes;
708    let lost_bytes = r.bbr2_state.newly_lost_bytes;
709
710    if lost_bytes > 0 {
711        // QUIC mininum cwnd is 2 x MSS.
712        r.congestion_window = r
713            .congestion_window
714            .saturating_sub(lost_bytes)
715            .max(r.max_datagram_size * recovery::MINIMUM_WINDOW_PACKETS);
716    }
717
718    if r.bbr2_state.packet_conservation {
719        r.congestion_window = r.congestion_window.max(in_flight + acked_bytes);
720    }
721}

723// 4.6.4.5.  Modulating cwnd in ProbeRTT
724fn bbr2_probe_rtt_cwnd(r: &mut Congestion) -> usize {
725    let probe_rtt_cwnd =
726        bbr2_bdp_multiple(r, r.bbr2_state.bw, PROBE_RTT_CWND_GAIN);
727
728    probe_rtt_cwnd.max(bbr2_min_pipe_cwnd(r))
729}
730
731fn bbr2_bound_cwnd_for_probe_rtt(r: &mut Congestion) {
732    if r.bbr2_state.state == BBR2StateMachine::ProbeRTT {
733        r.congestion_window = r.congestion_window.min(bbr2_probe_rtt_cwnd(r));
734    }
735}

737// 4.6.4.6.  Core cwnd Adjustment Mechanism
738fn bbr2_set_cwnd(r: &mut Congestion, in_flight: usize) {
739    let acked_bytes = r.bbr2_state.newly_acked_bytes;
740
741    bbr2_update_max_inflight(r);
742    bbr2_modulate_cwnd_for_recovery(r, in_flight);
743
744    if !r.bbr2_state.packet_conservation {
745        if r.bbr2_state.filled_pipe {
746            r.congestion_window = cmp::min(
747                r.congestion_window + acked_bytes,
748                r.bbr2_state.max_inflight,
749            )
750        } else if r.congestion_window < r.bbr2_state.max_inflight ||
751            r.delivery_rate.delivered() <
752                r.max_datagram_size * r.initial_congestion_window_packets
753        {
754            r.congestion_window += acked_bytes;
755        }
756
757        r.congestion_window = r.congestion_window.max(bbr2_min_pipe_cwnd(r))
758    }
759
760    bbr2_bound_cwnd_for_probe_rtt(r);
761    bbr2_bound_cwnd_for_model(r);
762}

764// 4.6.4.7.  Bounding cwnd Based on Recent Congestion
765fn bbr2_bound_cwnd_for_model(r: &mut Congestion) {
766    let mut cap = usize::MAX;
767
768    if bbr2_is_in_a_probe_bw_state(r) &&
769        r.bbr2_state.state != BBR2StateMachine::ProbeBWCRUISE
770    {
771        cap = r.bbr2_state.inflight_hi;
772    } else if r.bbr2_state.state == BBR2StateMachine::ProbeRTT ||
773        r.bbr2_state.state == BBR2StateMachine::ProbeBWCRUISE
774    {
775        cap = bbr2_inflight_with_headroom(r);
776    }
777
778    // Apply inflight_lo (possibly infinite).
779    cap = cap.min(r.bbr2_state.inflight_lo);
780    cap = cap.max(bbr2_min_pipe_cwnd(r));
781
782    r.congestion_window = r.congestion_window.min(cap);
783}
